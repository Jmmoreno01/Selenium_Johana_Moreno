import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.testng.annotations.Test;

import java.util.List;

import org.openqa.selenium.By;
public class EjercicioSelenium14 {

    @Test
    public void ejercicioNetflix(){
        System.setProperty("webdriver.chrome.driver","drivers/chromedriver.exe");
        WebDriver driver = new ChromeDriver();
        driver.get("https://www.netflix.com/co/");

        List<WebElement> listaH1s = driver.findElements(By.tagName("h1"));
        System.out.println("la cantidad de H1s del sitio son: " + listaH1s.size());

        List<WebElement> listaH2s = driver.findElements(By.tagName("h2"));
        System.out.println("la cantidad de H2s del sitio son: " +listaH2s.size());

        driver.navigate().refresh();

        List<WebElement> ListaBotones = driver.findElements(By.tagName("button"));
        System.out.print("La cantidad de botones de la página son : " + ListaBotones.size());

        for (WebElement boton:ListaBotones){
            System.out.println("\n"+boton.getText());
        }

        List<WebElement> div = driver.findElements(By.tagName("div"));
        System.out.print("La cantidad de etiquetas div son : " + div.size());

        System.out.println(" El titulo de la página es " + driver.getTitle());

        List<WebElement> ListaLinks = driver.findElements(By.tagName("a"));
        System.out.print("La cantidad de links son : " + ListaLinks.size());

        driver.close();
    }
}
